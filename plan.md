def push_markdown_to_github(slug, title, description, content_body):
    # Establish strict Markdown data pattern matching Astro's TypeScript Frontmatter

title: "{title}"
description: "{description}"
publishDate: "{datetime.now().strftime('%Y-%m-%d')}"

        "content": base64_string,
        "branch": "main"
    }
    if sha:
        payload["sha"] = sha

    put_res = requests.put(url, json=payload, headers=headers)
    if put_res.status_code in:
